198 static inline enum sleepmgr_mode sleepmgr_get_sleep_mode(void)
199 {
200  enum sleepmgr_mode sleep_mode = SLEEPMGR_ACTIVE;
201 
202 #ifdef CONFIG_SLEEPMGR_ENABLE
203  uint8_t *lock_ptr = sleepmgr_locks;
204 
205  // Find first non-zero lock count, starting with the shallowest modes.
206  while (!(*lock_ptr)) {
207  lock_ptr++;
208  sleep_mode = (enum sleepmgr_mode)(sleep_mode + 1);
209  }
210 
211  // Catch the case where one too many sleepmgr_unlock_mode() call has been
212  // performed on the deepest sleep mode.
213  Assert((uintptr_t)(lock_ptr - sleepmgr_locks) < SLEEPMGR_NR_OF_MODES);
214 
215 #endif /* CONFIG_SLEEPMGR_ENABLE */
216 
217  return sleep_mode;
218 }

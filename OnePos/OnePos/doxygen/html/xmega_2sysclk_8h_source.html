530 static inline uint32_t sysclk_get_per4_hz(void)
531 {
532  uint8_t shift = 0;
533 
534 #if XMEGA_E
535  if (CONFIG_SYSCLK_PSADIV > SYSCLK_PSADIV_512) {
536  switch (CONFIG_SYSCLK_PSADIV) {
537  case SYSCLK_PSADIV_6:
538  return sysclk_get_main_hz() / 6;
539  case SYSCLK_PSADIV_10:
540  return sysclk_get_main_hz() / 10;
541  case SYSCLK_PSADIV_12:
542  return sysclk_get_main_hz() / 12;
543  case SYSCLK_PSADIV_24:
544  return sysclk_get_main_hz() / 24;
545  case SYSCLK_PSADIV_48:
546  return sysclk_get_main_hz() / 48;
547  default:
548  //unhandled_case;
549  return 0;
550  }
551  }
552 #endif
553  if (CONFIG_SYSCLK_PSADIV & (1U << CLK_PSADIV_gp)) {
554  shift = (CONFIG_SYSCLK_PSADIV >> (1 + CLK_PSADIV_gp)) + 1;
555  }
556 
557  return sysclk_get_main_hz() >> shift;
558 }
